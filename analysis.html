merge_lang_cols <- function(df, langs = c("en", "de")) {
  pattern <- paste0("_(", paste(langs, collapse = "|"), ")$")
  
  lang_cols <- names(df)[stringr::str_detect(names(df), pattern)]
  bases <- stringr::str_remove(lang_cols, pattern)
  
  for (base in unique(bases)) {
    matching_cols <- lang_cols[bases == base]
    
    if (length(matching_cols) >= 2) {
      df[[base]] <- purrr::reduce(df[matching_cols], dplyr::coalesce)
      df <- df %>% dplyr::select(-dplyr::all_of(matching_cols))
    }
  }
  
  df
}

# Helper functions
stars <- function(p) {
  dplyr::case_when(
    p < .001 ~ "***",
    p < .01  ~ "**",
    p < .05  ~ "*",
    p < .10  ~ "†",
    TRUE ~ ""
  )
}
